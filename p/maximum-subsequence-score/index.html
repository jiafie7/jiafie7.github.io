class Solution
{
public:
    long long maxScore(vector<int>& nums1, vector<int>& nums2, int k)
    {
        using PII = std::pair<int, int>;
        std::vector<PII> combine;

        // Â∞Ü nums1 Âíå nums2 ÁöÑÂÖÉÁ¥†ÁªÑÂêàÊàê‰∫åÂÖÉÁªÑ
        for (int i = 0; i < nums1.size(); ++i)
            combine.push_back({nums2[i], nums1[i]});

        // Êåâ nums2 ÈôçÂ∫èÊéíÂàó
        std::sort(combine.rbegin(), combine.rend());

        long long nums1_sum = 0, max_score = 0;
        std::priority_queue<int, std::vector<int>, std::greater<int>> min_heap;

        // ÈÅçÂéÜÊØè‰∏™ÂÖÉÁ¥†ÔºåÁª¥Êä§‰∏Ä‰∏™Â§ßÂ∞è‰∏∫ k ÁöÑÂ≠êÂ∫èÂàó
        for (auto t : combine)
        {
            int num2 = t.first, num1 = t.second;
            nums1_sum += num1;
            min_heap.push(num1);

            // Â¶ÇÊûúÂ†ÜÂ§ßÂ∞èË∂ÖËøá kÔºåÂàôÁßªÈô§ÊúÄÂ∞èÂÄº
            if (min_heap.size() > k)
            {
                nums1_sum -= min_heap.top();
                min_heap.pop();
            }

            // Êõ¥Êñ∞ÊúÄÂ§ßÂàÜÊï∞
            if (min_heap.size() == k)
                max_score = std::max(max_score, nums1_sum * num2);
        }

        return max_score;
    }
};
