class Solution
{
public:
    void rotate(vector<vector<int>>& matrix)
    {
        int n = matrix.size();

        // 1. ËΩ¨ÁΩÆÁü©Èòµ
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < i; ++j)
                std::swap(matrix[i][j], matrix[j][i]);

        // 2. Ê∞¥Âπ≥ÁøªËΩ¨Áü©Èòµ
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n / 2; ++j)
                std::swap(matrix[i][j], matrix[i][n - 1 - j]);
    }
};
