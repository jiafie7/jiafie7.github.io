class Solution
{
public:
    vector<int> findAnagrams(string s, string p)
    {
        std::unordered_map<char, int> hash; // Â≠òÂÇ®ÁõÆÊ†áÂ≠óÁ¨¶È¢ëÁéá

            ++hash[c];

        std::vector<int> res;
        int total = hash.size(), count = 0;

        for (int i = 0, j = 0; i < s.size(); ++ i)
        {
            // Âè≥Êâ©ÔºöÂä†ÂÖ•Â≠óÁ¨¶ s[i]

        }

    }
