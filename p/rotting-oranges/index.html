class Solution
{
public:
    int orangesRotting(vector<vector<int>>& grid)
    {
        using PII = std::pair<int, int>;
        std::queue<PII> q;
        int n = grid.size(), m = grid[0].size();

        // Â∞ÜÊâÄÊúâÂàùÂßãËÖêÁÉÇÊ©òÂ≠êÁöÑÂùêÊ†áÂä†ÂÖ•ÈòüÂàó
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j)
                if (grid[i][j] == 2)
                    q.push({i, j});

        int res = 0;
        if (!q.empty()) --res; // ÈòüÂàóÈùûÁ©∫ÔºåÂàùÂßãÂåñÂàÜÈíüÊï∞‰∏∫ -1
        int dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

        // BFS Êâ©Êï£ËÖêÁÉÇ
        while (!q.empty()) {
            ++res; // ÊØèÈÅçÂéÜ‰∏ÄÂ±ÇÈòüÂàóÔºåÊó∂Èó¥Âä† 1
            int len = q.size();
            while (len--) {
                PII tmp = q.front();
                q.pop();
                int x = tmp.first, y = tmp.second;

                // ÈÅçÂéÜÂõõ‰∏™ÊñπÂêë
                for (int i = 0; i < 4; ++i) {
                    int a = x + dx[i], b = y + dy[i];
                    if (a < 0 || a >= n || b < 0 || b >= m || grid[a][b] != 1)
                        continue;
                    q.push({a, b});     // Âä†ÂÖ•Êñ∞ËÖêÁÉÇÁöÑÊ©òÂ≠ê
                    grid[a][b] = 2;     // Ê†áËÆ∞‰∏∫ËÖêÁÉÇ
                }
            }
        }

        // Ê£ÄÊü•ÊòØÂê¶ÊúâÂâ©‰ΩôÊñ∞È≤úÊ©òÂ≠ê
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j)
                if (grid[i][j] == 1)
                    return -1; // ‰ªçÊúâÊñ∞È≤úÊ©òÂ≠êÔºåËøîÂõû -1

        return res; // ËøîÂõûÊâÄÈúÄÁöÑÂàÜÈíüÊï∞
    }
