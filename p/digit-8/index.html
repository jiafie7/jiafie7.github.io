int bfs(std::string& start_state)
{
  std::string end_state = "12345678x";  // ÁõÆÊ†áÁä∂ÊÄÅ
  std::queue<std::string> q;  // BFS ÈòüÂàó
  std::unordered_map<std::string, int> d;  // ËÆ∞ÂΩïÊ≠•Êï∞

  q.push(start_state);
  d[start_state] = 0;

  // ÊñπÂêëÊï∞ÁªÑÔºöÂè≥„ÄÅ‰∏ã„ÄÅÂ∑¶„ÄÅ‰∏ä
  int dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};

  while (!q.empty())
  {
    std::string tmp = q.front();
    q.pop();

    int distance = d[tmp];  // ÂèñÂΩìÂâçÁä∂ÊÄÅÁöÑÊ≠•Êï∞

      return distance;  // ÁõÆÊ†áÁä∂ÊÄÅÔºåËøîÂõûÊúÄÂ∞èÊ≠•Êï∞

    int k = tmp.find('x');  // ÊâæÂà∞ 'x' ÁöÑÁ¥¢Âºï
    int x = k / 3, y = k % 3;  // ËΩ¨Êç¢‰∏∫‰∫åÁª¥ÂùêÊ†á

    for (int i = 0; i < 4; ++ i)  // ÈÅçÂéÜ 4 ‰∏™ÊñπÂêë
    {
      int a = x + dx[i], b = y + dy[i];
      if (a >= 0 && a < 3 && b >= 0 && b < 3)  // Á°Æ‰øùÊñ∞‰ΩçÁΩÆÂêàÊ≥ï
      {
        std::swap(tmp[k], tmp[a * 3 + b]);  // ‰∫§Êç¢ 'x' ‰∏éÁõ∏ÈÇªÊï∞Â≠ó

        if (!d.count(tmp))  // ËØ•Áä∂ÊÄÅÊú™ËÆøÈóÆ
        {
          q.push(tmp);  // Âä†ÂÖ•ÈòüÂàó
          d[tmp] = distance + 1;  // ËÆ∞ÂΩïÊ≠•Êï∞

        std::swap(tmp[a * 3 + b], tmp[k]);  // ‰∫§Êç¢ÂõûÊù•ÔºåÊÅ¢Â§çÂéüÁä∂ÊÄÅ
      }
    }
  }

  return -1;  // Êó†Ê≥ïÂà∞ËææÁõÆÊ†áÁä∂ÊÄÅ

  }

  return 0;
}
