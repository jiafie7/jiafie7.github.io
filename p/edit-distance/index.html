class Solution
{
public:
    int minDistance(string word1, string word2)
    {
        // Âú®Â≠óÁ¨¶‰∏≤ÂºÄÂ§¥Ê∑ªÂä†Á©∫Ê†ºÔºå‰æø‰∫éËæπÁïåÂ§ÑÁêÜ
        word1 = " " + word1;
        word2 = " " + word2;
        int n = word1.size(), m = word2.size();

        std::vector<std::vector<int>> f(n, std::vector<int>(m));

        // ÂàùÂßãÂåñËæπÁïåÊù°‰ª∂
        for (int i = 0; i < n; ++i)
            f[i][0] = i;
        for (int j = 0; j < m; ++j)
            f[0][j] = j;

        for (int i = 1; i < n; ++i)
            for (int j = 1; j < m; ++j)
            {
                // Âà†Èô§„ÄÅÊèíÂÖ•„ÄÅÊõøÊç¢Êìç‰ΩúÁöÑÊúÄ‰ºòÈÄâÊã©
                f[i][j] = std::min(f[i - 1][j] + 1, f[i][j - 1] + 1);
                if (word1[i] == word2[j])
                    f[i][j] = std::min(f[i][j], f[i - 1][j - 1]);
                else
                    f[i][j] = std::min(f[i][j], f[i - 1][j - 1] + 1);
            }

        // ËøîÂõûÁªìÊûú
        return f[n - 1][m - 1];
    }
};
