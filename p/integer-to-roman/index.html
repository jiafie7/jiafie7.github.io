class Solution
{
public:
    string intToRoman(int num)
    {
        // ÂÆö‰πâÊï∞ÂÄºÂíåÂØπÂ∫îÁöÑÁΩóÈ©¨Êï∞Â≠ó
        std::vector<int> values =
        {
            1000, 900, 500, 400, 100,
            90, 50, 40, 10, 9, 5, 4, 1
        };

        std::vector<std::string> romans =
        {
            "M", "CM", "D", "CD", "C",
            "XC", "L", "XL", "X", "IX", "V", "IV", "I"
        };

        std::string res;
        // ÈÅçÂéÜÊï∞ÂÄºÊï∞ÁªÑ
        for (int i = 0; i < values.size(); ++i)
        {
            // ÂáèÂéªÂΩìÂâçÂÄºÔºåÊãºÊé•ÂØπÂ∫îÁΩóÈ©¨Êï∞Â≠ó
            while (num >= values[i])
            {
                num -= values[i];
                res += romans[i];
            }
        }
        return res;
    }
};
