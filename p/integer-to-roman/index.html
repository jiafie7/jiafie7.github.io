class Solution
{
public:
    string intToRoman(int num)
    {
        // å®šä¹‰æ•°å€¼å’Œå¯¹åº”çš„ç½—é©¬æ•°å­—
        std::vector<int> values =
        {
            1000, 900, 500, 400, 100,
            90, 50, 40, 10, 9, 5, 4, 1
        };

        std::vector<std::string> romans =
        {
            "M", "CM", "D", "CD", "C",
            "XC", "L", "XL", "X", "IX", "V", "IV", "I"
        };

        std::string res;
        // éå†æ•°å€¼æ•°ç»„
        for (int i = 0; i < values.size(); ++i)
        {
            // å‡å»å½“å‰å€¼ï¼Œæ‹¼æ¥å¯¹åº”ç½—é©¬æ•°å­—
            while (num >= values[i])
            {
                num -= values[i];
                res += romans[i];
            }
        }
        return res;
    }
};
