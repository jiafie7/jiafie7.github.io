        int a = nums.top(); nums.pop();
        char c = op.top(); op.pop();
        if (c == '+')
            nums.push(a + b);
        else if (c == '-')
            nums.push(a - b);
        else if (c == '*')
            nums.push(a * b);
        else
            nums.push(a / b);

        {
            s.replace(pos, n, dest);
            pos = s.find(src);
