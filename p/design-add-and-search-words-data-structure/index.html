class WordDictionary
{
public:
    struct Node {
        Node() {
            ended = false;
            for (int i = 0; i < 26; ++i)
                son[i] = nullptr;
        }
        Node* son[26];  // å­èŠ‚ç‚¹æ•°ç»„
        bool ended;     // æ ‡è®°å•è¯ç»“æŸ
    };

    Node* root;

    WordDictionary() {
        root = new Node();
    }

    // æ’å…¥å•è¯
    void addWord(string word) {
        Node* p = root;
        for (char c : word) {
            int u = c - 'a';
            if (!p->son[u])
                p->son[u] = new Node();
            p = p->son[u];
        }
        p->ended = true;
    }

    // æŸ¥æ‰¾å•è¯
    bool search(string word) {
        return dfs(root, word, 0);
    }

    // æ·±åº¦ä¼˜å…ˆæœç´¢å®ç°æŸ¥æ‰¾é€»è¾‘
    bool dfs(Node* p, std::string word, int i) {
        if (i == word.size())
            return p->ended;  // æŠµè¾¾å•è¯æœ«å°¾ï¼Œåˆ¤æ–­æ˜¯å¦æœ‰æ•ˆ
        if (word[i] != '.') {
            int u = word[i] - 'a';
            if (p->son[u])
                return dfs(p->son[u], word, i + 1);
        } else {
            for (int j = 0; j < 26; ++j) {
                if (p->son[j] && dfs(p->son[j], word, i + 1))
                    return true;
            }
        }
        return false;
    }
};
